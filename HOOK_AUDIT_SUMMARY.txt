BUG #2: useNeuroTerminal.ts (Lines 36-54)
──────────────────────────────────────────
Severity: MEDIUM
Type: Stale Closure + Wrong Dependency Array
Fixed: YES ✓

Issue:
  - addLog callback has [logs.length] in dependencies
  - This causes callback to recreate on EVERY log addition
  - Also reads stale logs value inside callback
  - Defeats purpose of useCallback (memoization)

Before:
  const addLog = useCallback(
    (level, message, details?) => {
      const entry = { ... };
      logBufferRef.current.push(entry);
      setLogs((prev) => [...prev, entry]);

      // ❌ Reads stale logs from closure
      if (logs.length > 500) {
        setLogs((prev) => prev.slice(-500));
      }
    },
    [logs.length]  // ❌ Causes constant re-creation
  );

After:
  const addLog = useCallback(
    (level, message, details?) => {
      const entry = { ... };
      logBufferRef.current.push(entry);
      setLogs((prev) => {
        const updated = [...prev, entry];
        // ✅ Uses updater function (never stale)
        return updated.length > 500 ? updated.slice(-500) : updated;
      });
    },
    []  // ✅ No dependencies needed with updater
  );


BUG #3: useHeartbeat.ts (Lines 94-103)
───────────────────────────────────────
Severity: MEDIUM
Type: Unnecessary Circular Dependency Pattern
Fixed: YES ✓

Issue:
  - checkAll depends on captureSnapshot
  - captureSnapshot has no dependencies
  - Creates unnecessary dependency pattern
  - Can cause cascading effect re-runs in memoized components

Before:
  const checkAll = useCallback(async () => {
    await healthMonitor.checkAll();
    await captureSnapshot();  // ❌ Creates dependency
  }, [captureSnapshot]);  // ❌ Unnecessary

After:
  const checkAll = useCallback(async () => {
    await healthMonitor.checkAll();
    const log = await heartbeatMonitor.captureSnapshot();
    setSnapshot(log.snapshot);  // ✅ Inline the logic
  }, []);  // ✅ Fully stable callback


BUG #4: useQueuedPrompts.ts (Lines 107-118)
─────────────────────────────────────────────
Severity: HIGH
Type: Stale State + Unnecessary Dependency
Fixed: YES ✓

Issue:
  - deleteQueue reads selectedQueueId from closure
  - selectedQueueId is in dependency array
  - Callback recreates on every selection change
  - Can fail to clear selection when deleting selected queue

Before:
  const deleteQueue = useCallback(async (queueId: string) => {
    await QueuedPromptManager.deleteQueue(queueId);
    setQueues((prev) => prev.filter((q) => q.id !== queueId));

    // ❌ Reads stale selectedQueueId
    if (selectedQueueId === queueId) {
      setSelectedQueueId(null);
    }

    // ...
  }, [selectedQueueId]);  // ❌ Recreates on every selection

After:
  const deleteQueue = useCallback(async (queueId: string) => {
    await QueuedPromptManager.deleteQueue(queueId);
    setQueues((prev) => prev.filter((q) => q.id !== queueId));

    // ✅ Use updater function
    setSelectedQueueId((prev) => (prev === queueId ? null : prev));

    // ...
  }, []);  // ✅ Fully stable

  }, [subscribeToQueue]);  // ✅ Fixed
